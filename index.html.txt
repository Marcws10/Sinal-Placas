      <label for="placa">Placa:</label>
      <input type="text" id="placa" placeholder="Digite a placa (sem '-')">
    </div>
    <button onclick="adicionarSinal()">Alterar</button>
    <div id="output"></div>
  </div>

  <script>
    function adicionarSinal() {
      var placaInput = document.getElementById('placa');
      var placa = placaInput.value;
      
      if (placa.length >= 2) {
        var placaComSinal = placa.slice(0, 3) + '-' + placa.slice(3);
        document.getElementById('output').textContent = 'Placa com sinal: ' + placaComSinal;
      } else {
        document.getElementById('output').textContent = 'Placa inv√°lida.';
      }
    }
  </script>
